## 创建应用
在命令行窗口执行以下命令
```bash=
mvn archetype:generate -DarchetypeGroupId=com.casic.cplm -DarchetypeArtifactId=cplm-cloud-quickstart-archetype -DarchetypeVersion=1.0.0-SNAPSHOT
```
输入Maven项目的groupId、artifactId、version和默认包路径，如下所示
```tex=
Define value for property 'groupId': com.casic.cplm
Define value for property 'artifactId': cplm-cloud-demo
Define value for property 'version' 1.0-SNAPSHOT: : 1.0.0-SNAPSHOT
Define value for property 'package' com.casic.cplm: : com.casic.cplm.demo
```

### 创建Service
在cplm-cloud-demo-service/src/main/java/com/casic/cplm/demo/service创建接口文件OrderService.java和类OrderServiceImpl.java
```java=
package com.casic.cplm.demo.service;

import com.casic.cplm.demo.dto.OrderDto;
import com.casic.cplm.demo.entity.OrderEntity;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;

public interface OrderService {

  /**
   * 查询订单列表
   *
   * @param pageable 分页参数
   * @return 分页的订单列表
   */
  Page<OrderEntity> listOrders(Pageable pageable);

  /**
   * 创建订单
   *
   * @param form 创建表单信息
   * @return 订单对象
   */
  OrderEntity createOrder(OrderDto form);

  /**
   * 查询订单
   *
   * @param orderId 订单ID
   * @return 订单对象
   */
  OrderEntity getOrder(String orderId);

  /**
   * 修改订单
   *
   * @param orderId 订单ID
   * @param form 修改表单信息
   * @return 订单对象
   */
  OrderEntity modifyOrder(String orderId, OrderDto form);

  /**
   * 删除订单
   *
   * @param orderId 订单ID
   */
  void deleteOrder(String orderId);
}
```

```java=
package com.casic.cplm.demo.service;

import com.casic.cplm.demo.dto.OrderDto;
import com.casic.cplm.demo.dto.OrderQueryDto;
import com.casic.cplm.demo.entity.OrderEntity;
import com.casic.cplm.demo.repository.OrderRepository;
import java.util.ArrayList;
import java.util.List;
import javax.persistence.criteria.Predicate;
import org.apache.commons.lang3.StringUtils;
import org.apache.commons.lang3.math.NumberUtils;
import org.springframework.beans.BeanUtils;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.jpa.domain.Specification;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Service
@Transactional(rollbackFor = Exception.class)
public class OrderServiceImpl implements OrderService {
  private final OrderRepository orderRepository;

  public OrderServiceImpl(OrderRepository orderRepository) {
    this.orderRepository = orderRepository;
  }

  @Override
  public Page<OrderEntity> listOrders(OrderQueryDto query, Pageable pageable) {
    String name = StringUtils.trimToEmpty(query.getName());
    String description = StringUtils.trimToEmpty(query.getDescription());
    Specification<OrderEntity> specification =
        (root, criteriaQuery, cb) -> {
          List<Predicate> predicates = new ArrayList<>();
          // 按照名称过滤
          if (StringUtils.isNotBlank(name)) {
            predicates.add(cb.like(root.get("name"), "%" + name + "%"));
          }
          // 按照说明过滤
          if (StringUtils.isNotBlank(description)) {
            predicates.add(cb.like(root.get("description"), "%" + description + "%"));
          }
          return cb.and(predicates.toArray(new Predicate[0]));
        };
    return orderRepository.findAll(specification, pageable);
  }

  @Override
  public OrderEntity createOrder(OrderDto form) {
    OrderEntity entity = new OrderEntity();
    BeanUtils.copyProperties(form, entity);
    return orderRepository.save(entity);
  }

  @Override
  public OrderEntity getOrder(String orderId) {
    return orderRepository.findById(NumberUtils.toLong(orderId)).orElse(null);
  }

  @Override
  public OrderEntity modifyOrder(String orderId, OrderDto form) {
    OrderEntity entity = orderRepository.findById(NumberUtils.toLong(orderId)).orElse(null);
    if (entity == null) {
      throw new RuntimeException("订单不存在：" + orderId);
    }
    BeanUtils.copyProperties(form, entity);
    return orderRepository.save(entity);
  }

  @Override
  public void deleteOrder(String orderId) {
    orderRepository.findById(NumberUtils.toLong(orderId)).ifPresent(orderRepository::delete);
  }
}
```

### 创建Controller
在cplm-cloud-demo-starter/src/main/java/com/casic/cplm/demo/web创建类文件OrderController.java
> Controller中接口路径定义需要符合[REST](http://zh.wikipedia.org/wiki/REST)风格
```java=
package com.casic.cplm.demo.web;

import com.casic.cplm.data.EntityUtils;
import com.casic.cplm.demo.dto.OrderDto;
import com.casic.cplm.demo.dto.OrderQueryDto;
import com.casic.cplm.demo.entity.OrderEntity;
import com.casic.cplm.demo.service.OrderService;
import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.tags.Tag;
import java.util.List;
import java.util.stream.Collectors;
import org.springframework.beans.BeanUtils;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.repository.support.PageableExecutionUtils;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RestController;

@RestController
@Tag(name = "订单服务接口")
public class OrderController {

  private final OrderService orderService;

  @Autowired
  public OrderController(OrderService orderService) {
    this.orderService = orderService;
  }

  @Operation(summary = "订单列表")
  @GetMapping(value = "/orders")
  public ResponseEntity<Page<OrderDto>> listOrders(OrderQueryDto query, Pageable pageable) {
    Page<OrderEntity> page = orderService.listOrders(query, pageable);
    List<OrderDto> dtoList = page.stream().map(this::toDto).collect(Collectors.toList());
    return ResponseEntity.ok(
        PageableExecutionUtils.getPage(dtoList, pageable, page::getTotalElements));
  }

  @Operation(summary = "创建订单")
  @PostMapping(value = "/orders")
  public ResponseEntity<OrderDto> createOrder(@RequestBody OrderDto form) {
    form.setTenantId("0");
    OrderEntity entity = orderService.createOrder(form);
    return ResponseEntity.ok(toDto(entity));
  }

  @Operation(summary = "查询订单")
  @GetMapping(value = "/orders/{orderId}")
  public ResponseEntity<OrderDto> createOrder(@PathVariable String orderId) {
    OrderEntity entity = orderService.getOrder(orderId);
    return ResponseEntity.ok(toDto(entity));
  }

  @Operation(summary = "修改订单")
  @PutMapping(value = "/orders/{orderId}")
  public ResponseEntity<OrderDto> modifyOrder(
      @PathVariable String orderId, @RequestBody OrderDto form) {
    OrderEntity entity = orderService.modifyOrder(orderId, form);
    return ResponseEntity.ok(toDto(entity));
  }

  @Operation(summary = "修改订单")
  @DeleteMapping(value = "/orders/{orderId}")
  public ResponseEntity<Void> deleteOrder(@PathVariable String orderId) {
    orderService.deleteOrder(orderId);
    return ResponseEntity.noContent().build();
  }

  private OrderDto toDto(OrderEntity entity) {
    if (entity == null) {
      return null;
    }
    OrderDto dto = new OrderDto();
    BeanUtils.copyProperties(entity, dto);
    dto.setId(EntityUtils.getStringId(entity));
    return dto;
  }
}
```

## 支持的数据库
默认情况下，使用H2数据库作为演示使用，开发和测试过程中建议选择以下数据库
* MySQL 8
* Oracle 11g
* 达梦 8
### 使用MySQL数据库
1. 下载并安装MySQL数据库
2. 创建数据库和用户
```sql=
mysql -u root -p
CREATE DATABASE cplm DEFAULT CHARACTER SET utf8mb4;
CREATE USER 'cplm_user'@'%' IDENTIFIED BY 'cplm_password';
GRANT ALL ON cplm.* TO 'cplm_user'@'%';
exit
``` 
3. 修改cplm-cloud-demo-app/src/main/resources/application.properties文件
```profile=
spring.jpa.database-platform=org.hibernate.dialect.MySQL8Dialect
spring.jpa.generate-ddl=true
spring.jpa.open-in-view=false

spring.datasource.driver-class-name=com.mysql.cj.jdbc.Driver
spring.datasource.url=jdbc:mysql://localhost:3306/cplm?useUnicode=true&characterEncoding=utf8&useSSL=false&serverTimezone=UTC&allowPublicKeyRetrieval=true
spring.datasource.username=cplm_user
spring.datasource.password=cplm_password
```
### 使用Oracle数据库
1. 安装Oracle数据库
2. 创建数据库实例，名称为cplm
3. 创建数据库用户
```sql=
sqlplus /nolog
conn /as sysdba
create user cplm_user identified by cplm_password temporary tablespace TEMP default tablespace USERS;
grant connect, resource to cplm_user;
grant create sequence, create view, query rewrite, unlimited tablespace, select any dictionary to cplm_user;
``` 
4. 修改cplm-cloud-demo-app/src/main/resources/application.properties文件
```profile=
spring.jpa.database-platform=org.hibernate.dialect.Oracle10gDialect
spring.jpa.generate-ddl=true
spring.jpa.open-in-view=false

spring.datasource.driver-class-name=oracle.jdbc.driver.OracleDriver
spring.datasource.url=jdbc:oracle:thin:@localhost:1521:cplm
spring.datasource.username=cplm_user
spring.datasource.password=cplm_password
```
### 使用达梦数据库
1. 安装达梦数据库
2. 初始化数据库默认实例
3. 使用SYSDBA用户登录，创建数据库用户
```sql=
CREATE USER USER_NAME;
GRANT RESOURCE,VTI TO CPLM_USER;
```
4. 使用SYSSSO用户登录，设置数据库用户密码
```sql=
ALTER USER CPLM_USER IDENTIFIED BY "CPLM_PASSWORD123";
```
5. 修改cplm-cloud-demo-app/src/main/resources/application.properties文件
```profile=
spring.jpa.database-platform=org.hibernate.dialect.DmDialect
spring.jpa.generate-ddl=true
spring.jpa.open-in-view=false

spring.datasource.driver-class-name=dm.jdbc.driver.DmDriver
spring.datasource.url=jdbc:dm://localhost:5236?compatibleMode=oracle&dbmdChkPrv=false
spring.datasource.username=CPLM_USER
spring.datasource.password=CPLM_PASSWORD123
```
